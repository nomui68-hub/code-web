		{\allowdisplaybreaks \begin{eqnarray*}
				& &|z+w-u|=|u+z-w|\\
				&\Leftrightarrow& \left|\overrightarrow{OM}+\overrightarrow{ON}-\overrightarrow{OP}\right|=\left|\overrightarrow{OP}+\overrightarrow{OM}-\overrightarrow{ON}\right|\\
